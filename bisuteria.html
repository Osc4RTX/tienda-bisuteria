    <script>

        let carrito = [];


        function agregarAlCarrito(nombre, precio) {

            // Verifica si el producto ya está en el carrito para sumar la cantidad

            let productoExistente = carrito.find(item => item.nombre === nombre);

            if (productoExistente) {

                productoExistente.cantidad++;

            } else {

                carrito.push({ nombre: nombre, precio: precio, cantidad: 1 });

            }

            actualizarCarrito();

        }


        function actualizarCarrito() {

            const listaDiv = document.getElementById('lista-carrito');

            const totalSpan = document.getElementById('total-carrito');

            

            if (carrito.length === 0) {

                listaDiv.innerHTML = '<p>El carrito está vacío.</p>';

                totalSpan.innerText = '0.00';

                return;

            }


            listaDiv.innerHTML = '';

            let total = 0;


            carrito.forEach((item, index) => {

                total += item.precio * item.cantidad;

                listaDiv.innerHTML += `

                    <div class="carrito-item">

                        <span>${item.cantidad}x ${item.nombre}</span>

                        <span>$${(item.precio * item.cantidad).toFixed(2)}</span>

                    </div>

                `;

            });


            totalSpan.innerText = total.toFixed(2);

        }


        function enviarPedido() {

            if (carrito.length === 0) {

                alert("Por favor, agrega productos al carrito antes de enviar tu pedido.");

                return;

            }


            // === CAMBIA ESTE NÚMERO ===

            // Usa el código de país sin el + (Ej: 58 para Venezuela)

            const NUMERO_TIENDA = "584120000000"; 


            let mensaje = "¡Hola! Vengo de la página web y me gustaría hacer el siguiente pedido:\n\n";

            let total = 0;


            // Arma la lista de productos para el mensaje

            carrito.forEach(item => {

                mensaje += `▪️ ${item.cantidad}x ${item.nombre} ($${(item.precio * item.cantidad).toFixed(2)})\n`;

                total += item.precio * item.cantidad;

            });


            mensaje += `\n*Total a pagar: $${total.toFixed(2)}*\n\nPor favor, indíquenme los métodos de pago.`;


            // Codifica el texto para que la URL de WhatsApp no se rompa con los espacios y saltos de línea

            const urlWhatsApp = `https://wa.me/${NUMERO_TIENDA}?text=${encodeURIComponent(mensaje)}`;

            

            // Abre WhatsApp en una nueva pestaña

            window.open(urlWhatsApp, '_blank');

        }
